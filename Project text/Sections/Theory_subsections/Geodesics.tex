\begin{equation*}
    \gamma(t)\coloneqq \mathrm{exp}_{p}\big(t(Xp^{+}-(Xp^{+}) ^{\mathrm{T}})\big)\mathrm{exp}_{p}\big(t(Xp^{+})^{\mathrm{T}}\big)p,
\end{equation*}
